open System.IO

open System.Net.Http

open System.Net.Http.Headers

open System.Threading.Tasks

open FSharp.Json // Vagy FSharp.Data, ha inkább azt preferálod

open FSharp.Control.Tasks // Task-ok kényelmesebb kezeléséhez

 

// --- Konfiguráció ---

// **FONTOS: Cseréld le a 'YOUR_GITHUB_TOKEN' helyére a GitHub Personal Access Tokenedet!**

// A legjobb gyakorlat szerint ezt környezeti változóból vagy konfigurációs fájlból olvasnánk be.

let gitHubToken = "YOUR_GITHUB_TOKEN"

let gitHubApiBaseUrl = "https://api.github.com"

let userAgent = "FSharpGitHubCategorizerApp" // Szükséges a GitHub API hívásokhoz

 

// --- Adatmodellek (a GitHub API válaszának leképezése) ---

type GitHubRepository = {

    name: string

    full_name: string

    description: string option // Lehet null

    language: string option   // Lehet null

    topics: string list option // Lehet null, de mi listaként kezeljük

    stargazers_count: int

    html_url: string

}

 

// Ezek kellenek a GitHub API pagination (lapozás) kezeléséhez

type GitHubSearchResponse = {

    total_count: int

    incomplete_results: bool

    items: GitHubRepository list

}

 

// --- Kategória definíciók ---

// Itt definiálod a kategóriáidat a nyelvek és a témák alapján.

// Minden tuple: (Kategória neve, Nyelvek listája, Témák listája)

// A listák lehetnek üresek, ha csak a másik feltétel érdekes.

let projectCategories = [

    ("F# Projektek", ["F#"], [])

    ("Web Fejlesztés", ["JavaScript"; "TypeScript"; "HTML"; "CSS"; "C#"], ["web"; "frontend"; "backend"; "api"; "fullstack"])

    ("Gépi Tanulás/AI", ["Python"; "R"; "Julia"], ["machine-learning"; "ai"; "deep-learning"; "data-science"])

    ("Mobil Alkalmazások", ["Kotlin"; "Swift"; "Java"; "Objective-C"], ["android"; "ios"; "mobile"])

    ("Adatbázisok", [], ["database"; "sql"; "nosql"; "postgresql"; "mongodb"])

    ("DevOps/Infrastruktúra", [], ["devops"; "cloud"; "kubernetes"; "docker"; "terraform"])

    ("Egyéb", [], []) // Mindig legyen egy "egyéb" kategória a maradékhoz

]

 

// --- Segédfüggvények ---

let private httpClient =

    let client = new HttpClient()

    client.DefaultRequestHeaders.Add("User-Agent", userAgent)

    client.DefaultRequestHeaders.Add("Accept", "application/vnd.github.v3+json")

    client.DefaultRequestHeaders.Authorization <- AuthenticationHeaderValue("token", gitHubToken)

    client

 

let getRepositories (query: string) (page: int) : Task<GitHubSearchResponse option> = task {

    let url = $"{gitHubApiBaseUrl}/search/repositories?q={query}&per_page=100&page={page}"

    printfn "Lekérdezés: %s" url

    try

        let! response = httpClient.GetAsync(url)

        response.EnsureSuccessStatusCode() |> ignore

        let! jsonString = response.Content.ReadAsStringAsync()

        // Az FSharp.Json használatával deszerializáljuk

        let result = Json.parse<GitHubSearchResponse> jsonString

        return Some result

    with ex ->

        printfn "Hiba a GitHub API hívásakor: %s" ex.Message

        return None

}

 

// --- Projekt kategorizálása ---

let categorizeRepository (repo: GitHubRepository) =

    let mutable foundCategory = None

    for (categoryName, languages, topics) in projectCategories do

        let languageMatch =

            match repo.language with

            | Some lang -> languages |> List.contains lang

            | None -> languages |> List.isEmpty // Ha nincs nyelv megadva, és a kategória nem vár el nyelvet

 

        let topicsMatch =

            match repo.topics with

            | Some repoTopics ->

                // Ha van megadva téma a kategóriában, akkor a repo témái közül legalább egynek egyeznie kell

                if topics |> List.isEmpty then true // Ha a kategória nem vár el témát, akkor igaz

                else repoTopics |> List.exists (fun rt -> topics |> List.contains rt)

            | None -> topics |> List.isEmpty // Ha nincs téma a repo-ban, és a kategória nem vár el témát

 

        if languageMatch && topicsMatch then

            foundCategory <- Some categoryName

            break // Találtunk kategóriát, nem kell tovább keresni

   

    // Ha nem találtunk specifikus kategóriát, az "Egyéb" kategóriába kerül

    match foundCategory with

    | Some cat -> cat

    | None -> "Egyéb"

 

// --- Fõ program logika ---

[<EntryPoint>]

let main argv = task {

    if gitHubToken = "YOUR_GITHUB_TOKEN" then

        printfn "HIBA: Kérlek, cseréld le a 'YOUR_GITHUB_TOKEN' helyére a GitHub Personal Access Tokenedet a Program.fs fájlban!"

        return 1

 

    printfn "GitHub Projekt Kategorizáló indul..."

    printfn "-------------------------------------"

 

    // Példa: Keress F# projekteket vagy "web" témájú projekteket

    let searchQuery = "language:F# OR topic:web"

    // Vagy keress szervezeten belül (pl. Microsoft F#)

    // let searchQuery = "org:fsharp OR language:F#" // Példa: F# Software Foundation

    // let searchQuery = "org:dotnet" // Példa: .NET szervezet

 

    let mutable allRepositories = []

    let mutable currentPage = 1

    let mutable totalCount = -1

 

    printfn "Projektek lekérdezése a GitHubról (%s)..." searchQuery

 

    // Lapozás a lekérdezési eredményeken

    while (currentPage = 1 && totalCount = -1) || (currentPage > 1 && (float currentPage * 100.0) < (float totalCount)) do

        let! resultOpt = getRepositories searchQuery currentPage

 

        match resultOpt with

        | Some result ->

            totalCount <- result.total_count

            printfn "Lap: %d, Találatok ezen a lapon: %d, Összes találat: %d" currentPage result.items.Length totalCount

            allRepositories <- allRepositories @ result.items

            currentPage <- currentPage + 1

            do! Task.Delay(TimeSpan.FromSeconds(1.0)) // Kis szünet a rate limit elkerülésére

        | None ->

            printfn "Hiba történt a lekérdezés során, vagy nincs több találat."

            break

 

    printfn "-------------------------------------"

    printfn "Lekérdezve %d projekt." allRepositories.Length

    printfn "-------------------------------------"

 

    // Kategorizált projektek tárolása

    let categorizedProjects =

        allRepositories

        |> List.map (fun repo ->

            let category = categorizeRepository repo

            (category, repo)

        )

        |> List.groupBy fst // Csoportosítás kategória név szerint

 

    // Eredmények kiírása

    for (category, repos) in categorizedProjects do

        printfn "\n### Kategória: %s (%d projekt) ###" category repos.Length

        for (_, repo) in repos do

            printfn "  - %s (%s) - Nyelv: %A, Témák: %A, Csillagok: %d" repo.full_name repo.html_url repo.language repo.topics repo.stargazers_count

 

    printfn "-------------------------------------"

    printfn "Kategorizálás befejezõdött."

    return 0

}
